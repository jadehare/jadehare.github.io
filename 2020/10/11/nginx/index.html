Centos V2ray + Nginx + WS

Posted on 2020-10-11

1.yum -y update
2.yum -y install gcc pcre pcre-devel zlib zlib-devel openssl openssl-devel
3.yum -y install nginx

4.bash <(curl -L https://raw.githubusercontent.com/v2fly/fhs-install-v2ray/master/install-release.sh)

5.yum -y install git
6.git clone https://github.com/letsencrypt/letsencrypt
7../letsencrypt-auto certonly –standalone -d yoururl.com

8.修改nginx配置

server
{
listen 80;
listen 443 ssl http2;
ssl_certificate /etc/letsencrypt/live/yoururl.com/fullchain.pem;
ssl_certificate_key /etc/letsencrypt/live/yoururl.com/privkey.pem;
ssl_protocols TLSv1 TLSv1.1 TLSv1.2 TLSv1.3;
ssl_ciphers TLS13-AES-256-GCM-SHA384:TLS13-CHACHA20-POLY1305-SHA256:TLS13-AES-128-GCM-SHA256:TLS13-AES-128-CCM-8-SHA256:TLS13-AES-128-CCM-SHA256:EECDH+CHACHA20:EECDH+AES128:RSA+AES128:EECDH+AES256:RSA+AES256:EECDH+3DES:RSA+3DES:!MD5;
ssl_prefer_server_ciphers on;
ssl_session_timeout 10m;
ssl_session_cache builtin:1000 shared:SSL:10m;
ssl_buffer_size 1400;
add_header Strict-Transport-Security max-age=15768000;
ssl_stapling on;
ssl_stapling_verify on;
server_name yoururl.com;
access_log /etc/nginx/log/yoururl.com_nginx.log combined;
index index.html index.htm index.php;
root /data/wwwroot/yoururl.com;
if ($ssl_protocol = “”) { return 301 https://$host$request_uri; }

#error_page 404 /404.html;
#error_page 502 /502.html;

location /v2ray { // 自订
proxy_pass http://127.0.0.1:9999;
proxy_redirect off;
proxy_http_version 1.1;
proxy_set_header Upgrade $http_upgrade;
proxy_set_header Connection “upgrade”;
proxy_set_header Host $http_host;
}
}

9.添加v2ray 配置

touch /etc/v2ray/config.json

{

“inbounds”: [
{
“port”: 9999,
“listen”:”127.0.0.1”,
“protocol”: “vmess”,
“settings”: {
“clients”: [
{
“id”: “8e1f77d1-82bc-4278-a5d5-a1e134385626”, // 生成自己的id
“level”: 1,
“alterId”: 64
}
]
},
“streamSettings”: {
“network”: “ws”,
“wsSettings”: {
“path”: “/v2ray” // 自订，同nginx配置
}
}
}
],
“dns”: {
“servers”: [“8.8.8.8”, “localhost”]
},
“outbounds”: [
{
“protocol”: “freedom”,
“settings”: {}
},
{
“protocol”: “blackhole”,
“settings”: {},
“tag”: “blocked”
}
],
“routing”: {
“rules”: [
{
“type”: “field”,
“ip”: [“geoip:private”],
“outboundTag”: “blocked”
}
]
}
}

nohup v2ray -config /etc/v2ray/config.json 2>&1 &
